// Test 1: Basic features - Variables, expressions, and simple control structures
// Prueba características básicas - Variables, expresiones y estructuras de control simples

// Variable declarations (with and without semicolons)
var x : Int = 10;
val y : Int = 20
var z : Float = 3.14f;
var name : String = "Hello"
var flag : Boolean = true;

// Expressions and assignments
var sum : Int = x + y
var diff : Int = y - x;
var product : Int = x * y
z *= 2.0f;
x += 5

// Logical and comparison expressions
var isEqual : Boolean = x == y
var isLess : Boolean = x < y;
var andResult : Boolean = flag && (x > 0)
var orResult : Boolean = flag || false;
var notResult : Boolean = !flag

// Print statements (with and without semicolons)
println("=== Basic Test Results ===")
print("x = "); println(x);
print("sum = "); println(sum)
print("isLess = "); println(isLess);

// Simple if statements
if (x > y) {
    println("x is greater")
} else {
    println("y is greater")
}

// Single statement if (no braces)
if (flag)
    println("flag is true")
    
// Simple while loop
var counter : Int = 0
while (counter < 3) {
    print("count: "); println(counter)
    counter += 1;
}

// Simple for loop
println("For loop:")
for (i in 1..3) {
    println(i)
}

// Blocks with local scope
{
    var localVar : Int = 100;
    println("In block:");
    println(localVar)
}

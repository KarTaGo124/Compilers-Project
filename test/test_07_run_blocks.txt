// Test 7: Run blocks
fun main(): Unit {
    // Run blocks bÃ¡sicos
    run {
        var runVar: Int = 42
        println("Inside run block: " + runVar)
        
        run {
            var nestedRunVar: String = "Nested run"
            println(nestedRunVar)
            
            if (runVar > 40) {
                println("Nested run condition true")
            }
        }
        
        for (runLoop in 1..2) {
            println("Run block loop: " + runLoop)
        }
    }
    
    println("Run blocks test passed!")
}

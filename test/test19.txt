fun main(): Unit {
    var threshold: Float = 5.5f
    var value: Float = 1.0f
    var sum: Float = 0.0f

    while (value < threshold) {
        if (value > 2.5f) {
            sum += value * 1.5f
        } else {
            sum += value / 2.0f
        }

        value += 0.75f

        if (sum > 10.0f) {
            break
        }
    }

    var counter: Float = 10.0f
    do {
        sum -= counter / 3.0f
        counter -= 1.25f
    } while (counter > 5.0f && sum > 0.0f)
    println(sum)

    println("Test floats control-estructuras completado")
}
